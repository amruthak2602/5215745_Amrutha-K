 /*
 * Complete the 'truckTour' function below.
 *
 * The function is expected to return an INTEGER.
 * The function accepts 2D_INTEGER_ARRAY petrolpumps as parameter.
 */
 int truckTour(int petrolpumps_rows, int petrolpumps_columns, int** 
petrolpumps) {
 long long total_petrol = 0;
 long long curr_petrol = 0;
 int start_index = 0;
 for (int i = 0; i < petrolpumps_rows; i++) {
 int petrol = petrolpumps[i][0];
 int distance = petrolpumps[i][1];
        total_petrol += (petrol - distance);
        curr_petrol += (petrol - distance);
        if (curr_petrol < 0) {
            // Cannot start from this pump or any before it
            start_index = i + 1;
            curr_petrol = 0;
        }
    }
    if (total_petrol >= 0) {
        return start_index;
    } else {
        return -1; // Should not happen as per problem guarantees
    }
}
